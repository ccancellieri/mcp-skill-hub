{# Plugin extension-point: A11 — shared data table macro. `columns` is a list of column definitions: either strings (header label, used as both key and label) or dicts {key, label, align?}. `rows` is a list of dicts keyed by column.key. #} {% macro data_table(columns, rows, empty_message='No data') -%} {% set cols = [] %} {% for c in columns %} {% if c is string %} {% set _ = cols.append({'key': c, 'label': c, 'align': 'left'}) %} {% else %} {% set _ = cols.append({'key': c.key, 'label': c.label or c.key, 'align': c.align or 'left'}) %} {% endif %} {% endfor %} {% for c in cols %}{% endfor %} {% if rows %} {% for row in rows %} {% for c in cols %}{% endfor %} {% endfor %} {% else %} {% endif %}
{{ c.label }}
{{ row[c.key] }}
{{ empty_message }}
{%- endmacro %}
